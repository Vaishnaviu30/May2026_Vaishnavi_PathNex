
Ansible — Install + Start Nginx
---
- name: Install and start nginx
  hosts: all
  become: yes

  tasks:
    - name: Install nginx
      yum:
        name: nginx
        state: present

    - name: Start and enable nginx
      service:
        name: nginx
        state: started
        enabled: yes

Kubernetes — Deployment (3 Replicas)
apiVersion: apps/v1
kind: Deployment

metadata:
  name: nginx-deployment

spec:
  replicas: 3

  selector:
    matchLabels:
      app: nginx

  template:
    metadata:
      labels:
        app: nginx

    spec:
      containers:
      - name: nginx
        image: nginx

## Jenkins Pipeline — Deploy to Multiple Environments
pipeline {
    agent any
    environment {
        INSTITUTE_NAME = "Pathnex"
    }
    stages {
        stage('Deploy to Dev') {
            steps {
                sh 'echo "Deploying $INSTITUTE_NAME to Dev environment"'
            }
        }
        stage('Deploy to Staging') {
            steps {
                input message: "Approve Staging Deployment?"
                sh 'echo "Deploying $INSTITUTE_NAME to Staging environment"'
            }
        }
        stage('Deploy to Prod') {
            steps {
                input message: "Approve Prod Deployment?"
                sh 'echo "Deploying $INSTITUTE_NAME to Prod environment"'
            }
        }
    }
}

## GitLab CI — Multi-Environment
